IEnumerable扩展

	
		1、 All(是否都为真)
		
			if (list.All(obj => obj.Name.Equals("b")))
			{
			}
		
		2、 Any(无参：有元素返真，有参，有满足的元素返真)
		
			if (list.Any(obj => obj.Id.Equals("1")))
			{
			}
		
		3、 AsEnumerable(复制)
		
		    list.AsEnumerable();
		
		4、 Concat(合并两个集合)
		
		    list.Concat(list2);
		
		5、 Contains(是否包含)
		
			if (list.Contains(p))
			{
			}
		
		6、 Count(数量，有参：条件过滤)
		
	        int num = list.Count(obj => obj.Age.Equals(10));
		
	    7、 Distinct(返回非重复项)
		
	    	var item = list.Distinct();
		
		8、 ElementAt(返回指定索引值)
		
	    	var item = list.ElementAt(0);
		
		9、 ElementAtOrDefault(返回指定索引值或NULL)
	 
	        var item = list.ElementAtOrDefault(3);
		
		10、First(返回第一个值)
		
	    	var item = list.First(obj=>obj.Age.Equals(10));
		
		11、FirstOrDefault(返回第一个值或NULL)
		
	    	var item = list.FirstOrDefault(obj => obj.Age.Equals(100));
		
		12、GroupBy(分组)
		
		13、GroupJoin(比较两个集合)
		
		14、Last(返回最后一项)
		
	    	var item = list.Last(obj => obj.Age.Equals(10));
		
		15、LastOrDefault(返回最后一项或NULL)      
		
	    	var item = list.LastOrDefault(obj => obj.Age.Equals(30));
		
		16、LongCount(Count升级版)
		
		17、Max(最大值)
		
	    	var item = list.Max(obj => obj.Age);
		
		18、Min(最小值)
		
	    	var item = list.Min(obj => obj.Age);
	 	
		19、OrderBy(升序排序)
		
	    	var item = list.OrderBy(obj => obj.Age).ToList();
		
		20、OrderByDescending(降序排序)
		
	    	var item = list.OrderByDescending(obj => obj.Age).ToList();
		
		21、Reverse(集合反转)
		
	    	list.Reverse();
		
		22、Exists(是否存在)
		
			if (list.Exists(obj => obj.Id == "21"))
