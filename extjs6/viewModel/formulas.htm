 formulas: {
     xy: function (get) {  // "get" is arbitrary but a good convention
         return get('x') * get('y');
     },

     xyz: function (get) {
         return get('xy') * get('z');
     }
 }
 formulas: {
        fullName: {
            get: function (get) {
                var ret = get('firstName') || '';

                if (get('lastName')) {
                    ret += ' ' +  get('lastName');
                }

                return ret;
            },

            set: function (value) {
                var space = value.indexOf(' '),
                    split = (space < 0) ? value.length : space;

                this.set({
                    firstName: value.substring(0, split),
                    lastName: value.substring(split + 1)
                });
            }
        }
    }
