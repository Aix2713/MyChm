 formulas: {
        fullName: {
            get: function (get) {
                var ret = get('firstName') || '';

                if (get('lastName')) {
                    ret += ' ' +  get('lastName');
                }

                return ret;
            },

            set: function (value) {
                var space = value.indexOf(' '),
                    split = (space < 0) ? value.length : space;

                this.set({
                    firstName: value.substring(0, split),
                    lastName: value.substring(split + 1)
                });
            }
        }
    }
