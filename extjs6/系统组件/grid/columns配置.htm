columns类型及配置格式

activecolumn：

        {
            xtype:'actioncolumn',
            width:50,
            items: [{
                iconCls: 'x-fa fa-cog',
                tooltip: 'Edit',
                handler: function(grid, rowIndex, colIndex) {
                    var rec = grid.getStore().getAt(rowIndex);
                    alert("Edit " + rec.get('firstname'));
                }
            },{
                icon: 'extjs-build/examples/restful/images/delete.png',
                tooltip: 'Delete',
                handler: function(grid, rowIndex, colIndex) {
                    var rec = grid.getStore().getAt(rowIndex);
                    alert("Terminate " + rec.get('firstname'));
                }
            }]
        }

booleancolumn：

        {
            xtype: 'booleancolumn',
            text: 'Rocks',
            trueText: 'Yes',
            falseText: 'No',
            dataIndex: 'rocks'
        }

note：rocks的值为true｜false

checkcolumn：

{ xtype: 'checkcolumn', text: 'Active', dataIndex: 'active' }

note：'active' 的值为true｜false

templatecolumn：

{text: 'Department (Yrs)', xtype:'templatecolumn', tpl:'{dep} ({seniority})'}

note：dep和seniority的值为dataIndex绑定的值

datecolumn：

        { text: 'Symbol',   dataIndex: 'symbol', flex: 1 },
        { text: 'Date',     dataIndex: 'date',   xtype: 'datecolumn',   format:'Y-m-d' },
        { text: 'Change',   dataIndex: 'change', xtype: 'numbercolumn', format:'0.00' },
        { text: 'Volume',   dataIndex: 'volume', xtype: 'numbercolumn', format:'0,000' },
        { text: 'Top Day',  dataIndex: 'topday', xtype: 'datecolumn',   format:'l' }

note：format格式化日期，具体参考日期格式

numbercolumn：

{ text: 'Change',         dataIndex: 'change', xtype: 'numbercolumn', format:'0.00' },
{ text: 'Volume',         dataIndex: 'volume', xtype: 'numbercolumn', format:'0,000' }

widgetcolumn：

{
        text: '%Capacity',
        width: 150,

        // This is our Widget column
        xtype: 'widgetcolumn',

        // This is the widget definition for each cell.
        // The Progress widget class's defaultBindProperty is 'value'
        // so its "value" setting is taken from the ViewModel's record "capacityUsed" field
        // Note that a row ViewModel will automatically be injected due to the existence of
        // the bind property in the widget configuration.
        widget: {
            xtype: 'progressbarwidget',
            bind: '{record.capacityUsed}',
            textTpl: [
                '{percent:number("0")}% capacity'
            ]
        }
