var combo = Ext.getCmp('roleStore');
console.log(combo)
if (combo == undefined) return value;
var index = combo.store.find(combo.valueField, value);
if (index < 0) return value;
var record = combo.store.getAt(index);
return record.data.roleName;
},
editor: {
completeOnEnter: true,
xtype: 'combo',
id: 'roleStore',
fieldLabel: '',
store: { type: 'role' },
