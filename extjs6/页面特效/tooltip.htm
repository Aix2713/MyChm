鼠标移动至页面，显示自定义提示

 

 

var tip = Ext.create('Ext.tip.ToolTip', {
    target: self.el,
    // constrainTo: this.getTargetEl(),
    closable: false,
    trackMouse: true,
    delegate: self.itemSelector,
    showDelay: 200,
    items: [{
        xtype: 'panel',
        width: 200,
        height: 200
    }],
    listeners: {
        // Change content dynamically depending on which element triggered the show.
        beforeshow: function(tip) {
            tip.down('panel').setTitle(self.getRecord(tip.triggerElement).id);
            console.log(self.getRecord(tip.triggerElement));
          }
    }
});
