大多数 REST api 都倾向于需要版本控制, 并且在接受版本的报头中支持 semver 版本控制的 restify 船舶, 与您指定 NPM 版本依赖项的方法相同:

var restify = require('restify');

var server = restify.createServer();

function sendV1(req, res, next) {
  res.send('hello: ' + req.params.name);
  return next();
}

function sendV2(req, res, next) {
  res.send({hello: req.params.name});
  return next();
}

var PATH = '/hello/:name';
server.get({path: PATH, version: '1.1.3'}, sendV1);
server.get({path: PATH, version: '2.0.0'}, sendV2);

server.listen(8080);

在第一种情况下, 我们根本没有指定接受版本的报头, 所以 restify 对待它就像发送一个 *。如果不发送接受标头, 就意味着客户端会得到服务器的选择。Restify 将选择这一最高匹配路线。在第二种情况下, 我们明确要求 V1, 它得到了我们的响应从版本1处理函数的响应, 但后来我们要求 V2, 并得到了 JSON。最后, 我们要求一个不存在的版本, 并得到了一个错误。

您可以通过在服务器创建时传入版本字段来默认在路由上的版本。最后, 您可以使用数组支持 API 中的多个版本:

server.get({path: PATH, version: ['2.0.0', '2.1.0', '2.2.0']}, sendV2);

在这种情况下, 您可能需要了解更多的信息, 如原始请求的版本字符串是什么, 以及从路由支持的版本数组中匹配的版本是什么。两种方法使此信息可用:

var PATH = '/version/test';
server.get({
  path: PATH,
  version: ['2.0.0', '2.1.0', '2.2.0']
}, function (req, res, next) {
  res.send(200, {
    requestedVersion: req.version(),
    matchedVersion: req.matchedVersion()
  });
  return next();
});

Hitting this route will respond as below:

$ curl -s -H 'accept-version: <2.2.0' localhost:8080/version/test | json
{
  "requestedVersion": "<2.2.0",
  "matchedVersion": "2.1.0"
}
